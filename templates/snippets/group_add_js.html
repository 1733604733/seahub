$("#group-add").click(function() { $("#group-add-form").modal({appendTo: "#main"}); }); $('#group-add-form').submit(function() { var submit_btn = $(this).children('input[type="submit"]'); submit_btn.attr('disabled', 'disabled'); $.ajax({ url: '{{ post_url }}', type: 'POST', dataType: 'json', cache: 'false', beforeSend: prepareCSRFToken, data: { 'group_name': $('#group_name').val() }, success: function(data) { if (data['success']) { location.reload(true); } else { apply_form_error('group-add-form', data['error']); submit_btn.removeAttr('disabled'); } }, error: function(data, textStatus, jqXHR) { var errors = $.parseJSON(data.responseText); $.each(errors, function(index, value) { if (index == 'error') { apply_form_error('group-add-form', value); } else { apply_form_error('group-add-form', value[0]); } }); submit_btn.removeAttr('disabled'); } }); return false; });
